/**
 * Calculates the Softmax probabilities for a set of raw score logits.
 * Ensures the outputs always sum to exactly 100%.
 */
function softmax(logits) {
    const maxLogit = Math.max(...logits); // To prevent floating-point overflow
    const exps = logits.map(l => Math.exp(l - maxLogit));
    const sumExps = exps.reduce((a, b) => a + b, 0);
    
    // Convert to percentages and round to nearest integer
    const percentages = exps.map(e => Math.round((e / sumExps) * 100));
    
    // Core Adjustment Constraint: Handle potential rounding errors so sum = 100
    const sumPct = percentages.reduce((a, b) => a + b, 0);
    if (sumPct !== 100) {
        const diff = 100 - sumPct;
        // Find the index of the highest probability to absorb the rounding error
        const maxIdx = percentages.indexOf(Math.max(...percentages));
        percentages[maxIdx] += diff;
    }
    
    return percentages;
}
